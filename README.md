## Install

```bash
npm install --save use-signal
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'use-signal'
import 'use-signal/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
